    state: {
    count: 0,

    },
    decrement() {
        const { count } = model();
        setState({ count: count - 1 });
    },
    async incrementAsync() {

        increment();
    },
    }),
};

function Counter() {

        <button onClick={decrement}>-</button>
        <button onClick={incrementAsync}>+ async{incrementAsync.loading && '...'}</button>
    </>
    );
}
